### 函数重载

```javascript
let addMethod = (object, name, fn) => {
  
  //保存原有的函数，因为调用的时候可能不匹配传入的参数个数
  let old = object[name];
  
  //创建一个新匿名函数作为新方法
  object[name] = (...args) => {
    
    //如果该匿名函数的形参个数和实参个数匹配，就调用该函数
    if (fn.length === args.length) {
      return fn.apply(this, args)
      
    //如果传入的参数不匹配，则调用原有的参数
    } else if (typeof old === 'function') {
      return old.apply(this, args);
    }
  }
}
```
