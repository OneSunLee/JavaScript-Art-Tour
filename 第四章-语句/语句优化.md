一层又一层的 `if` 语句，看起来有十分冗长。由于所有的判断条件都是比较相等，我们想到了将 `if` 语句改写为 `switch` 语句，这样能省略所有重复的相等表达式。

```javascript
const province = prompt("请输入省级行政区的完整名称");
let capital;

switch (province) {
case "北京市":
	capital = "北京"; break;
case "上海市":
	capital = "上海"; break;
case "天津市":
	capital = "天津"; break;
case "重庆市":
	capital = "重庆"; break;

	capital = "哈尔滨"; break;

	capital = "长春"; break;

	capital = "沈阳"; break;
case "内蒙古":

	capital = "昆明"; break;

	capital = "拉萨"; break;

	capital = "杭州"; break;

	capital = "南昌"; break;
case "广东省":
	capital = "广州"; break;
case "福建省":
	capital = "福州"; break;
case "台湾省":
	capital = "台北"; break;
case "海南省":
	capital = "海口"; break;

	capital = "香港"; break;

	capital = "澳门"; break;
default:
	capital = "未知";
}
```
